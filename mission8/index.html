Exercice 1
Demander 3 valeurs a l'utilisateur, en faire la somme et la moyenne, et renvoyer le résultat dans une nouvelle page web.

let n1 = prompt("Saisir nombre 1");
let n2 = prompt("Saisir nombre 2");
let n3 = prompt("Saisir nombre 3");

let somme = Number(n1) + Number(n2) + Number(n3);
let moyenne = somme/3;

let fenetre = window.open();

fenetre.document.write("Voici la somme : " + somme + " et voici la moyenne :" + moyenne);

Exercice 2
Changer l'arrière plan de la page web.

function demo_ex2(){
let couleur = prompt('Entrer rouge, bleu ou vert en minuscules');
if (couleur=='rouge'){
document.body.style.backgroundColor = 'red';}
else if (couleur=='bleu'){
document.body.style.backgroundColor = 'blue';}
else if(couleur=='vert'){
document.body.style.backgroundColor = 'green';}
else {
document.body.style.backgroundColor = 'beige';}
}
